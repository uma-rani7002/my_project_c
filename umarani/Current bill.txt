#include <stdio.h>

int main()
{
    float u,cost;
    printf("enter no.of units(u):");
    scanf("%f",&u);
    if(u<=50)
        printf("cost is %f",cost=u*(0.50));
    else if(u>50&&u<=150)
        printf("cost is %f",cost=u*(0.75));
    else if(u>150&&u<=250)
        printf("cost is %f",cost=u*(1.20));
    else
        printf("cost is %f",cost=u*(1.50));
}
